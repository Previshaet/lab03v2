cmake_minimum_required(VERSION 3.4)

set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

project(formatter_lib)

include_directories(
	../formatter_lib/
	../formatter_ex_lib/
	../solver_lib/
)

add_library(formatter_lib STATIC ../formatter_lib/formatter.cpp ../formatter_lib/formatter.h)
add_library(formatter_ex_lib STATIC ../formatter_ex_lib/formatter_ex.cpp ../formatter_ex_lib/formatter_ex.h)
add_library(solver_lib STATIC ../solver_lib/solver.cpp ../solver_lib/solver.h)	

add_executable(solver equation.cpp)

target_link_libraries(formatter_ex_lib formatter_lib)
target_link_libraries(solver solver_lib formatter_ex_lib)
